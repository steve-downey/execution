23template <::beman::execution::scope_token Token, ::beman::execution::sender Sender>
24struct associate_data {
25 using wrap_sender = ::std::remove_cvref_t<decltype(::std::declval<Token&>().wrap(::std::declval<Sender>()))>;
26
27 explicit associate_data(Token t, Sender&& s) : token(t), sender(this->token.wrap(::std::forward<Sender>(s))) {
28 if (!token.try_associate()) {
29 this->sender.reset();
30 }
31 }
32 associate_data(const associate_data& other) noexcept(::std::is_nothrow_copy_constructible_v<wrap_sender> &&
33 noexcept(token.try_associate()))
34 : token(other.token), sender() {
35 if (other.sender && this->token.try_associate()) {
36 try {
37 this->sender.emplace(*other.sender);
38 } catch (...) {
39 this->token.disassociate();
40 }
41 }
42 }
43 associate_data(associate_data&& other) noexcept(::std::is_nothrow_move_constructible_v<wrap_sender>)
44 : token(other.token), sender(::std::move(other.sender)) {
45 other.sender.reset();
46 }
47 auto operator=(const associate_data&) -> associate_data& = delete;
48 auto operator=(associate_data&&) -> associate_data& = delete;
49 ~associate_data() {
50 if (this->sender) {
51 this->sender.reset();
52 this->token.disassociate();
53 }
54 }
55
56 auto release() -> ::std::optional<::std::pair<Token, wrap_sender>> {
57 return this->sender ? (std::unique_ptr<std::optional<wrap_sender>, decltype([](auto* opt) { opt->reset(); })>(
58 &this->sender),
59 ::std::optional{::std::pair{::std::move(this->token), ::std::move(*this->sender)}})
60 : ::std::optional<::std::pair<Token, wrap_sender>>{};
61 }
62
63 Token token;
64 ::std::optional<wrap_sender> sender;
65};
66template <::beman::execution::scope_token Token, ::beman::execution::sender Sender>
67associate_data(Token, Sender&&) -> associate_data<Token, Sender>;

69struct associate_t {
70 template <::beman::execution::sender Sender, ::beman::execution::scope_token Token>
71 auto operator()(Sender&& sender, Token&& token) const {
72 auto domain(::beman::execution::detail::get_domain_early(sender));
73 return ::beman::execution::transform_sender(
74 domain,
75 ::beman::execution::detail::make_sender(
76 *this,
77 ::beman::execution::detail::associate_data(::std::forward<Token>(token),
78 ::std::forward<Sender>(sender))));
79 }

94 static constexpr auto get_state =
95 []<typename Sender, typename Receiver>(Sender&& sender, Receiver& receiver) noexcept(
96 ::std::is_nothrow_constructible_v<::std::remove_cvref_t<Sender>, Sender>&&
97 get_noexcept<::std::remove_cvref_t<Sender>, Receiver>::value) {
98 auto [_, data] = ::std::forward<Sender>(sender);
99 auto dataParts{data.release()};
100
101 using scope_token = decltype(dataParts->first);
102 using wrap_sender = decltype(dataParts->second);
103 using op_t = decltype(::beman::execution::connect(::std::move(dataParts->second),
104 ::std::forward<Receiver>(receiver)));
105
106 struct op_state {
107 using sop_t = op_t;
108 using sscope_token = scope_token;
109 struct assoc_t {
110 sscope_token tok;
111 sop_t op;
112 };
113
114 bool associated{false};
115 union {
116 Receiver* rcvr;
117 assoc_t assoc;
118 };
119 explicit op_state(Receiver& r) noexcept : rcvr(::std::addressof(r)) {}
120 explicit op_state(sscope_token tk, wrap_sender&& sndr, Receiver& r) try
121 : associated(true), assoc(tk, ::beman::execution::connect(::std::move(sndr), ::std::move(r))) {
122 } catch (...) {
123 tk.disassociate();
124 throw;
125 }
126 op_state(op_state&&) = delete;
127 ~op_state() {
128 if (this->associated) {
129 this->assoc.op.~sop_t();
130 this->assoc.tok.disassociate();
131 this->assoc.tok.~sscope_token();
132 }
133 }
134 auto run() noexcept -> void {
135 if (this->associated) {
136 ::beman::execution::start(this->assoc.op);
137 } else {
138 ::beman::execution::set_stopped(::std::move(*this->rcvr));
139 }
140 }
141 };
142 return dataParts ? op_state(::std::move(dataParts->first), ::std::move(dataParts->second), receiver)
143 : op_state(receiver);
144 };
145 static constexpr auto start = [](auto& state, auto&&) noexcept -> void { state.run(); };
146};
